常用基本操作

  • 2020.05.25

安装

Mysql社区下载地址

操作

  1. 更改root密码
mysqladmin -uroot password 'yourpassword'
  1. 远程登陆mysql服务器
mysql -uroot -p -h192.168.137.10 -P3306
  1. 查询数据库
show databases;
  1. 进入某个数据库
use databasename;
  1. 列出数据库中的表
show tables;
  1. 查看某个表全部字段
desc slow_log;
# 不仅可以显示表信息，还可以显示建表语句
show create table slow_log\G;
  1. 查看当前用户
select user();
  1. 查看当前所在数据库
select database();
  1. 创建新数据库（可以指定字符集）
create database db1 charset utf8;
  1. 创建新表
create table t1 (`id` int(4), `name` char(40));
  1. 查看数据库版本
select version();
  1. 查看数据库状态
show status;         # 当前会话状态
show global status;  # 全局数据库状态
show slave status\G;   # 查看主从数据库状态信息
  1. 查询数据库参数
show variables;
  1. 修改数据库参数
show variables like 'max_connect%';
set global max_connect_errors = 1000; # 重启数据库会失效，要在配置文件中修改
  1. 查看当前数据库队列
show processlist;
  1. 创建普通用户并授权给某个数据库
grant all on databasename.* to 'user1'@'localhost' identified by '123456';
  1. 查询表数据
select * from mysql.db;           # 查询该表中的所有字段
select count(*) from mysql.user;  # count(*)表示表中有多少行
select db,user  from mysql.db;    # 查询表中的多个字段
select * from mysql.db where host like '10.0.%'; # 在查询语句中可以使用万能匹配 “%”
  1. 插入一行数据
insert into db1.t1 values (1, 'abc');
insert into db1.t1 set id = 22;
  1. 更改表的某一行数据
update db1.t1 set name='aaa' where id=1;
  1. 清空表数据
truncate table db1.t1;
  1. 删除表
drop table db1.t1;
  1. 清空数据库中的所有表（数据库名是eab12）
mysql -N -s information_schema -e "SELECT CONCAT('TRUNCATE TABLE ',TABLE_NAME,';') FROM TABLES WHERE TABLE_SCHEMA='eab12'" | mysql -f eab12
  1. 删除数据库
drop database db1;
  1. 数据库备份
mysqldump  -uroot -p'yourpassword' mysql >/tmp/mysql.sql
  1. 数据库恢复
mysql -uroot -p'yourpassword' mysql </tmp/mysql.sql
  1. 新建普通用户
CREATE USER name IDENTIFIED BY 'ssapdrow';
  1. 更改普通用户密码
SET PASSWORD FOR name=PASSWORD('fdddfd');
  1. 查看name用户权限
SHOW GRANTS FOR name;
  1. 脚本中执行mysql命令
mysql -uuser -ppasswd -e"show databases"
echo "show databases"|mysql -uuser -ppassword
# 以下是执行大量mysql语句采用的方式
mysql -uuser -hhostname -ppasswd <<EOF
mysql语句
EOF
